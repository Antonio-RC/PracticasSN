Apendices.

Funcion Graf_continuo
function [] = graf_continuo(dominio,rango,titulo)
plot(dominio,rango,'b',"LineWidth",3)
grid on
axis([min(dominio) max(dominio) 0 1.25])
title(titulo)
end
Funcion graf_discreto.
function [] = graf_discreto(dominio,rango,titulo)
stem(dominio,rango,'g',"LineWidth",3)
grid on
axis([min(dominio) max(dominio) 0 1.25])
title(titulo)
end
Función de membresía 1-Ejercicio 10
function [y] = miu1(x, params)
%params(1) = k;
%params(2) = m;
y = 1./(1 + params(1)*(x-params(2)).^2);
end
Función de membresía 2-Ejercicio 10
function [y] = miu1(x, params)
%params(1) = k;
%params(2) = m;
y = 1./(1 + params(1)*(x-params(2)).^2);
end

function [y] = singleton(x,m)
%UNTITLED3 Summary of this function goes here
% Detailed explanation goes here
y = x;
for i = 1:length(x)
if x(i) == m
y(i) = 1;
else
y(i) = 0;
end
end
